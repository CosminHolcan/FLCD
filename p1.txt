p1 - verify if a number is prime

declare int a;
readFromInput(a);
declare int i = 3;
ifTrue (a == 2)
{
    return true;
}
ifTrue ((a < 2) or (a % 2 == 0))
{
    return false;
}
whileCondition (i*i < a) do
{
    ifTrue (a % i == 0)
    {
        return false;
    }
    i = i + 2;
}

return true;
